# 阶段八集成示例与回归记录

## 1. 集成示例（Theme + i18n + config + System-UX）

```tsx
import {
  ChipsThemeProvider,
  ChipsLoadingBoundary,
  ChipsNotification,
  ChipsErrorBoundary
} from "@chips/component-library";

const configSource = {
  systemUx: {
    loadingBoundary: { delayMs: 120, skeletonLines: 4 },
    notification: { maxVisible: 3, defaultDurationMs: 2800 }
  }
};

const i18n = {
  translate({ key }) {
    const dict = {
      "systemUx.loadingBoundary.status": "加载中",
      "systemUx.notification.close": "关闭通知"
    };
    return dict[key] || "";
  }
};

export function Demo() {
  return (
    <ChipsThemeProvider themeId="chips-official.default-theme" version="1.0.0">
      <ChipsErrorBoundary i18n={i18n} configSource={configSource}>
        <ChipsLoadingBoundary loading i18n={i18n} configSource={configSource}>
          <ChipsNotification i18n={i18n} configSource={configSource} defaultItems={[]} />
        </ChipsLoadingBoundary>
      </ChipsErrorBoundary>
    </ChipsThemeProvider>
  );
}
```
