## 1. 概述

## 2. 配置数据模型

### 2.1 卡片配置 (RichTextCardConfig)

存储位置：复合卡片文件的 `content/{ID}.yaml`

#### YAML Schema

```yaml
# JSON Schema for RichTextCard Config
$schema: "http://json-schema.org/draft-07/schema#"
title: "RichTextCard Configuration Schema"
type: object
required:
  - card_type
  - content_source

properties:
  card_type:
    const: "RichTextCard"
    description: "卡片类型标识，固定值"
    
  theme:
    type: string
    default: ""
    description: "主题包标识，格式：发行商:主题名。为空使用上级主题"
    example: "薯片官方:默认主题"
    
  layout:
    type: object
    description: "布局参数"
    properties:
      height_mode:
        type: string
        enum: ["auto", "fixed"]
        default: "auto"
        description: "高度模式：auto-自适应内容，fixed-固定高度"
      fixed_height:
        type: integer
        minimum: 50
        maximum: 10000
        description: "固定高度值（像素），仅height_mode=fixed时有效"
        
  content_source:
    type: string
    enum: ["file", "inline"]
    description: "内容来源：file-引用外部文件，inline-内联存储"
    
  content_file:
    type: string
    description: "富文本文件路径，相对于卡片根目录。content_source=file时必需"
    example: "content.html"
    
  content_text:
    type: string
    description: "内联HTML内容。content_source=inline时必需"
    example: "<p>Hello World</p>"
    
  toolbar:
    type: boolean
    default: false
    description: "是否显示编辑工具栏"
    
  read_only:
    type: boolean
    default: true
    description: "是否为只读模式"
```

#### TypeScript 类型定义

```typescript
/**
 * 富文本卡片配置
 * 保存在复合卡片的 content/{ID}.yaml 文件中
 */
interface RichTextCardConfig {
  /** 卡片类型标识，固定为 "RichTextCard" */
  card_type: 'RichTextCard';
  
  /** 主题包标识，为空使用上级主题 */
  theme?: string;
  
  /** 布局参数 */
  layout?: RichTextLayoutConfig;
  
  /** 内容来源 */
  content_source: 'file' | 'inline';
  
  /** 富文本文件路径（content_source=file时使用） */
  content_file?: string;
  
  /** 内联HTML内容（content_source=inline时使用） */
  content_text?: string;
  
  /** 是否显示工具栏 */
  toolbar?: boolean;
  
  /** 是否只读 */
  read_only?: boolean;
}

/**
 * 布局配置
 */
interface RichTextLayoutConfig {
  /** 高度模式 */
  height_mode?: 'auto' | 'fixed';
  
  /** 固定高度值（像素） */
  fixed_height?: number;
}
```

#### 配置示例

**内联模式**：
```yaml
card_type: "RichTextCard"
content_source: "inline"
content_text: |
  <h1>我的笔记</h1>
  <p>这是一段<strong>重要</strong>的内容。</p>
  <ul>
    <li>第一点</li>
    <li>第二点</li>
  </ul>
toolbar: false
read_only: true
```

**文件模式**：
```yaml
card_type: "RichTextCard"
content_source: "file"
content_file: "document.html"
layout:
  height_mode: "fixed"
  fixed_height: 400
toolbar: false
read_only: true
```

### 2.2 配置验证规则

| 规则 | 说明 |
|-----|------|
| card_type必须为"RichTextCard" | 用于识别卡片类型 |
| content_source必需 | 必须指定内容来源 |
| content_source=file时content_file必需 | 文件模式需要指定文件路径 |
| content_source=inline时content_text必需 | 内联模式需要提供内容 |
| fixed_height范围50-10000 | 防止极端值 |

## 3. 状态数据模型

### 3.1 渲染器状态 (RichTextRendererState)

```typescript
/**
 * 渲染器内部状态

interface RichTextRendererState {

  isLoading: boolean;

```

### 3.2 编辑器状态 (RichTextEditorState)

```typescript
/**
 * 编辑器内部状态
 */
interface RichTextEditorState {
  /** 当前HTML内容 */
  content: string;
  
  /** 当前选区 */
  selection: SelectionRange | null;
  
  /** 当前生效的格式集合 */
  activeFormats: Set<FormatType>;
  
  /** 当前光标所在块级元素类型 */
  currentBlock: BlockType;
  
  /** 是否可以撤销 */
  canUndo: boolean;
  
  /** 是否可以重做 */
  canRedo: boolean;
  
  /** 是否有未保存的修改 */
  isDirty: boolean;
  
  /** 字数统计 */
  wordCount: number;
  
  /** 是否聚焦 */
  isFocused: boolean;
}

/**
 * 选区范围
 */
interface SelectionRange {
  /** 起始偏移 */
  startOffset: number;
  
  /** 结束偏移 */
  endOffset: number;
  
  /** 是否折叠（光标无选区） */
  collapsed: boolean;
}

/**
 * 格式类型
 */
type FormatType = 
  | 'bold' 
  | 'italic' 
  | 'underline' 
  | 'strikethrough'
  | 'superscript'
  | 'subscript'
  | 'code';

/**
 * 块级元素类型
 */
type BlockType = 
  | 'paragraph'
  | 'heading1'
  | 'heading2'
  | 'heading3'
  | 'heading4'
  | 'heading5'
  | 'heading6'
  | 'orderedList'
  | 'unorderedList'
  | 'blockquote';
```

## 4. 命令数据模型

### 4.1 格式化命令 (FormatCommand)

```typescript
/**
 * 格式化命令
 * 用于执行文本格式化操作
 */
type FormatCommand =
  | BoldCommand
  | ItalicCommand
  | UnderlineCommand
  | StrikethroughCommand
  | HeadingCommand
  | OrderedListCommand
  | UnorderedListCommand
  | BlockquoteCommand
  | ColorCommand
  | BackgroundColorCommand
  | FontSizeCommand
  | AlignCommand
  | ClearFormatCommand;

interface BoldCommand {
  type: 'bold';
}

interface ItalicCommand {
  type: 'italic';
}

interface UnderlineCommand {
  type: 'underline';
}

interface StrikethroughCommand {
  type: 'strikethrough';
}

interface HeadingCommand {
  type: 'heading';
  /** 标题级别：1-6，0表示普通段落 */
  level: 0 | 1 | 2 | 3 | 4 | 5 | 6;
}

interface OrderedListCommand {
  type: 'orderedList';
}

interface UnorderedListCommand {
  type: 'unorderedList';
}

interface BlockquoteCommand {
  type: 'blockquote';
}

interface ColorCommand {
  type: 'color';
  /** 颜色值（HEX格式） */
  value: string;
}

interface BackgroundColorCommand {
  type: 'backgroundColor';
  /** 背景色值（HEX格式） */
  value: string;
}

interface FontSizeCommand {
  type: 'fontSize';
  /** 字号（像素） */
  value: number;
}

interface AlignCommand {
  type: 'align';
  /** 对齐方式 */
  value: 'left' | 'center' | 'right' | 'justify';
}

interface ClearFormatCommand {
  type: 'clearFormat';
}
```

### 4.2 插入命令 (InsertCommand)

```typescript
/**
 * 插入命令
 * 用于插入链接、图片等内容
 */
type InsertCommand =
  | InsertLinkCommand
  | InsertImageCommand
  | InsertHorizontalRuleCommand;

interface InsertLinkCommand {
  type: 'link';
  /** 链接URL */
  url: string;
  /** 链接文本（可选，默认使用选中文本或URL） */
  text?: string;
  /** 是否在新窗口打开 */
  newWindow?: boolean;
}

interface InsertImageCommand {
  type: 'image';
  /** 图片URL或相对路径 */
  src: string;
  /** 替代文本 */
  alt?: string;
  /** 宽度（可选） */
  width?: number;
  /** 高度（可选） */
  height?: number;
}

interface InsertHorizontalRuleCommand {
  type: 'horizontalRule';
}
```

## 5. 历史记录数据模型

interface HistoryEntry {

interface RenderOptions {

interface EditorOptions {

  theme?: string;

  locale?: string;
  
  /** 是否显示工具栏 */
  toolbar?: boolean;

 * 验证结果
