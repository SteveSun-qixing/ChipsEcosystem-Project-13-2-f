## 2. 插件主接口

### 2.1 RichTextCardPlugin

```typescript
/**
 * 富文本基础卡片插件主类
 * 
 * @description 插件入口类，负责管理插件生命周期和创建组件实例
 * 
 * @example
 * ```typescript
 * const plugin = new RichTextCardPlugin();
 * await plugin.initialize(core);
 * 
 * const renderer = plugin.createRenderer();
 * await renderer.render(config, container, options);
 * ```
 */
class RichTextCardPlugin implements BaseCardPlugin {
  /**
   * 插件元数据
   */
  readonly metadata: {
    /** 插件ID，格式：发行商:插件名 */
    id: 'chipshub:rich-text-card';
    /** 显示名称 */
    name: string;
    /** 版本号 */
    version: string;
    /** 支持的卡片类型 */
    cardType: 'RichTextCard';
    /** 图标路径 */
    icon: string;
    /** 插件描述 */
    description: string;
  };
  
  /**
   * 配置Schema（JSON Schema格式）
   */
  readonly configSchema: JSONSchema;
  
  /**
   * 初始化插件
   * 
   * @param core - 薯片内核实例
   * @returns Promise 初始化完成
   * 
   * @description
   * 1. 保存内核引用
   * 2. 加载依赖资源
   * 3. 注册插件服务
   */
  initialize(core: ChipsCore): Promise<void>;
  
  /**
   * 启动插件
   * 
   * @returns Promise 启动完成
   */
  start(): Promise<void>;
  
  /**
   * 停止插件
   * 
   * @returns Promise 停止完成
   */
  stop(): Promise<void>;
  
  /**
   * 销毁插件
   * 
   * @returns Promise 销毁完成
   * 
   * @description 清理所有资源，解绑服务
   */
  destroy(): Promise<void>;
  
  /**
   * 创建渲染器实例
   * 
   * @returns 新的渲染器实例
   */
  createRenderer(): RichTextRenderer;
  
  /**
   * 创建编辑器实例
   * 
   * @returns 新的编辑器实例
   */
  createEditor(): RichTextEditor;
}
```

## 3. 渲染组件接口

### 3.1 RichTextRenderer

```typescript
/**
 * 富文本渲染器
 * 
 * @description 负责将富文本配置渲染为可视化内容
 * 
 * @example
 * ```typescript
 * const renderer = plugin.createRenderer();
 * 
 * await renderer.render(config, container, {
 *   mode: 'view',
 *   theme: 'default'
 * });
 * 
 * // 更新内容
 * await renderer.update({ content_text: '<p>New content</p>' });
 * 
 * // 销毁
 * await renderer.destroy();
 * ```
 */
interface RichTextRenderer {
  /**
   * 渲染富文本内容
   * 
   * @param config - 卡片配置
   * @param container - 渲染目标容器
   * @param options - 渲染选项
   * @returns Promise 渲染完成
   * 
   * @throws {ChipsError} 配置无效时抛出 RICHTEXT-1001
   * @throws {ChipsError} 加载文件失败时抛出 RICHTEXT-2002
   */
  render(
    config: RichTextCardConfig,
    container: HTMLElement,
    options: RenderOptions
  ): Promise<void>;
  
  /**
   * 更新配置并重新渲染
   * 
   * @param config - 部分配置更新
   * @returns Promise 更新完成
   */
  update(config: Partial<RichTextCardConfig>): Promise<void>;
  
  /**
   * 获取当前状态
   * 
   * @returns 渲染器状态
   */
  getState(): RichTextRendererState;
  
  /**
   * 设置状态
   * 
   * @param state - 部分状态更新
   */
  setState(state: Partial<RichTextRendererState>): void;
  
  /**
   * 销毁渲染器
   * 
   * @returns Promise 销毁完成
   */
  destroy(): Promise<void>;
}
```

### 3.2 RenderOptions

```typescript
/**
 * 渲染选项
 */
interface RenderOptions {
  /**
   * 渲染模式
   * - view: 查看模式，只读展示
   * - edit: 编辑模式，可编辑
   */
  mode: 'view' | 'edit';
  
  /**
   * 主题包ID
   * 覆盖配置中的theme字段
   */
  theme?: string;
  
  /**
   * 是否只读
   * 覆盖配置中的read_only字段
   */
  readonly?: boolean;
  
  /**
   * 是否可交互
   * 控制链接点击、图片放大等交互
   * @default true
   */
  interactive?: boolean;
  
  /**
   * 语言环境
   * @default 'zh-CN'
   */
  locale?: string;
}
```

## 4. 编辑组件接口

### 4.1 RichTextEditor

```typescript
/**
 * 富文本编辑器
 * 
 * @description 提供所见即所得的富文本编辑能力
 * 
 * @example
 * ```typescript
 * const editor = plugin.createEditor();
 * 
 * await editor.render(config, container, {
 *   toolbar: true,
 *   autoSave: true
 * });
 * 
 * // 监听变更
 * editor.onChange((newConfig) => {
 *   console.log('Content changed:', newConfig);
 * });
 * 
 * // 执行格式化
 * editor.format({ type: 'bold' });
 * 
 * // 插入链接
 * editor.insert({ type: 'link', url: 'https://example.com' });
 * ```
 */
interface RichTextEditor {
  /**
   * 渲染编辑器
   * 
   * @param config - 卡片配置
   * @param container - 渲染目标容器
   * @param options - 编辑器选项
   * @returns Promise 渲染完成
   */
  render(
    config: RichTextCardConfig,
    container: HTMLElement,
    options: EditorOptions
  ): Promise<void>;
  
  /**
   * 获取当前配置
   * 
   * @returns 完整的卡片配置
   */
  getConfig(): RichTextCardConfig;
  
  /**
   * 设置配置
   * 
   * @param config - 部分配置更新
   */
  setConfig(config: Partial<RichTextCardConfig>): void;
  
  /**
   * 验证配置
   * 
   * @returns 验证结果
   */
  validate(): ValidationResult;
  
  /**
   * 注册变更回调
   * 
   * @param callback - 变更时调用的函数
   */
  onChange(callback: (config: RichTextCardConfig) => void): void;
  
  /**
   * 执行格式化命令
   * 
   * @param command - 格式化命令
   * 
   * @example
   * ```typescript
   * // 加粗
   * editor.format({ type: 'bold' });
   * 
   * // 设置标题
   * editor.format({ type: 'heading', level: 2 });
   * 
   * // 设置颜色
   * editor.format({ type: 'color', value: '#FF0000' });
   * ```
   */
  format(command: FormatCommand): void;
  
  /**
   * 插入内容
   * 
   * @param command - 插入命令
   * 
   * @example
   * ```typescript
   * // 插入链接
   * editor.insert({ 
   *   type: 'link', 
   *   url: 'https://example.com',
   *   text: '示例链接'
   * });
   * 
   * // 插入图片
   * editor.insert({
   *   type: 'image',
   *   src: 'image.jpg',
   *   alt: '示例图片'
   * });
   * ```
   */
  insert(command: InsertCommand): void;
  
  /**
   * 撤销上一步操作
   */
  undo(): void;
  
  /**
   * 重做上一步撤销
   */
  redo(): void;
  
  /**
   * 是否可以撤销
   * 
   * @returns true表示可以撤销
   */
  canUndo(): boolean;
  
  /**
   * 是否可以重做
   * 
   * @returns true表示可以重做
   */
  canRedo(): boolean;
  
  /**
   * 获取编辑器状态
   * 
   * @returns 编辑器状态
   */
  getState(): RichTextEditorState;
  
  /**
   * 聚焦编辑器
   */
  focus(): void;
  
  /**
   * 失去焦点
   */
  blur(): void;
  
  /**
   * 获取当前HTML内容
   * 
   * @returns HTML字符串
   */
  getContent(): string;
  
  /**
   * 设置HTML内容
   * 
   * @param html - HTML字符串
   */
  setContent(html: string): void;
  
  /**
   * 获取纯文本内容
   * 
   * @returns 纯文本字符串
   */
  getPlainText(): string;
  
  /**
   * 销毁编辑器
   * 
   * @returns Promise 销毁完成
   */
  destroy(): Promise<void>;
}
```

## 9. 使用示例

### 9.1 基础渲染

```typescript
import { RichTextCardPlugin } from '@chips/rich-text-card-plugin';

// 初始化插件
const plugin = new RichTextCardPlugin();
await plugin.initialize(core);

// 创建渲染器
const renderer = plugin.createRenderer();

// 配置
const config: RichTextCardConfig = {
  card_type: 'RichTextCard',
  content_source: 'inline',
  content_text: '<h1>Hello</h1><p>World</p>'
};

// 渲染
const container = document.getElementById('card-container');
await renderer.render(config, container, {
  mode: 'view',
  interactive: true
});
```

### 9.2 编辑器使用

```typescript
// 创建编辑器
const editor = plugin.createEditor();

// 渲染编辑器
await editor.render(config, container, {
  toolbar: true,
  autoSave: true,
  saveDelay: 500,
  placeholder: '请输入内容...'
});

// 监听变更
editor.onChange((newConfig) => {
  // 保存到卡片文件
  await core.request({
    service: 'card.saveConfig',
    payload: { cardId, config: newConfig }
  });
});

// 格式化操作
document.getElementById('bold-btn').onclick = () => {
  editor.format({ type: 'bold' });
};

// 插入链接
document.getElementById('link-btn').onclick = () => {
  const url = prompt('请输入链接地址');
  if (url) {
    editor.insert({ type: 'link', url });
  }
};

// 撤销/重做
document.getElementById('undo-btn').onclick = () => editor.undo();
document.getElementById('redo-btn').onclick = () => editor.redo();
```

### 9.3 与编辑引擎集成

```typescript
// 编辑引擎中使用

// 1. 用户点击富文本卡片
async function onCardSelected(cardId: string, cardType: string) {
  if (cardType !== 'RichTextCard') return;
  
  // 2. 加载插件
  const plugin = await core.getPlugin('chipshub:rich-text-card');
  
  // 3. 读取配置
  const config = await core.request({
    service: 'card.readConfig',
    payload: { cardId }
  });
  
  // 4. 创建编辑器
  const editor = plugin.createEditor();
  
  // 5. 渲染到编辑面板
  const editPanel = document.getElementById('edit-panel');
  await editor.render(config, editPanel, {
    toolbar: true,
    autoSave: true
  });
  
  // 6. 监听变更并保存
  editor.onChange(async (newConfig) => {
    await core.request({
      service: 'card.saveConfig',
      payload: { cardId, config: newConfig }
    });
  });
}
```
