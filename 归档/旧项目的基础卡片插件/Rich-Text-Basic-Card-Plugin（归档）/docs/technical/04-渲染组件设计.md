## 1. 概述

渲染组件负责将富文本配置数据渲染为可视化的HTML内容。本文档详细描述渲染组件的设计和实现。

## 3. 渲染器主类实现

### 3.1 RichTextRenderer.ts

```typescript
import { createApp, App } from 'vue';
import type { ChipsCore } from '@chips/core';
import type { 
  RichTextCardConfig, 
  RichTextRendererState,
  RenderOptions 
} from '../types';
import RichTextRendererVue from './RichTextRenderer.vue';
import { sanitizeHtml } from '../utils/sanitizer';

/**
 * 富文本渲染器
 */
export class RichTextRenderer {
  private core: ChipsCore | null = null;
  private config: RichTextCardConfig | null = null;
  private container: HTMLElement | null = null;
  private options: RenderOptions | null = null;
  private vueApp: App<Element> | null = null;
  private state: RichTextRendererState = {
    content: '',
    isLoading: false,
    error: null,
    currentTheme: '',
    containerWidth: 0
  };

  /**
   * 设置Core引用
   */
  setCore(core: ChipsCore): void {
    this.core = core;
  }

  /**
   * 渲染富文本内容
   */
  async render(
    config: RichTextCardConfig,
    container: HTMLElement,
    options: RenderOptions
  ): Promise<void> {
    this.config = config;
    this.container = container;
    this.options = options;
    
    // 设置加载状态
    this.setState({ isLoading: true, error: null });
    
    try {
      // 1. 加载内容
      const rawContent = await this.loadContent();
      
      // 2. 安全过滤
      const safeContent = this.sanitizeContent(rawContent);
      
      // 3. 更新状态
      this.setState({ 
        content: safeContent, 
        isLoading: false 
      });
      
      // 4. 挂载Vue组件
      this.mountVueComponent();
      
      // 5. 应用主题
      const themeId = options.theme || config.theme || '';
      await this.applyTheme(themeId);
      
      // 6. 绑定事件
      this.bindEvents();
      
    } catch (error) {
      this.setState({ 
        isLoading: false, 
        error: (error as Error).message 
      });
      throw error;
    }
  }

  /**
   * 更新配置
   */
  async update(config: Partial<RichTextCardConfig>): Promise<void> {
    if (!this.config) return;
    
    this.config = { ...this.config, ...config };
    
    // 如果内容变化，重新加载
    if (config.content_source || config.content_file || config.content_text) {
      const rawContent = await this.loadContent();
      const safeContent = this.sanitizeContent(rawContent);
      this.setState({ content: safeContent });
    }
    
    // 如果主题变化，重新应用
    if (config.theme !== undefined) {
      await this.applyTheme(config.theme);
    }
  }

  /**
   * 获取状态
   */
  getState(): RichTextRendererState {
    return { ...this.state };
  }

  /**
   * 设置状态
   */
  setState(state: Partial<RichTextRendererState>): void {
    this.state = { ...this.state, ...state };
  }

  /**
   * 销毁渲染器
   */
  async destroy(): Promise<void> {
    this.unbindEvents();
    
    if (this.vueApp) {
      this.vueApp.unmount();
      this.vueApp = null;
    }
    
    if (this.container) {
      this.container.innerHTML = '';
      this.container = null;
    }
    
    this.config = null;
    this.options = null;
  }

  /**
   * 加载内容
   */
  private async loadContent(): Promise<string> {
    if (!this.config) return '';
    
    if (this.config.content_source === 'inline') {
      return this.config.content_text || '';
    }
    
    if (this.config.content_source === 'file' && this.config.content_file) {
      // 通过Core读取文件
      const response = await this.core?.request({
        service: 'resource.read',
        payload: {
          uri: this.config.content_file,
          encoding: 'utf-8'
        }
      });
      
      if (!response?.success) {
        throw new Error(`Failed to load content file: ${this.config.content_file}`);
      }
      
      return response.data.content;
    }
    
    return '';
  }

  /**
   * 安全过滤内容
   */
  private sanitizeContent(html: string): string {
    return sanitizeHtml(html);
  }

  /**
   * 挂载Vue组件
   */
  private mountVueComponent(): void {
    if (!this.container) return;
    
    // 清空容器
    this.container.innerHTML = '';
    
    // 创建挂载点
    const mountPoint = document.createElement('div');
    mountPoint.className = 'chips-richtext-container';
    this.container.appendChild(mountPoint);
    
    // 创建Vue应用
    this.vueApp = createApp(RichTextRendererVue, {
      content: this.state.content,
      options: this.options,
      onLinkClick: this.handleLinkClick.bind(this),
      onImageClick: this.handleImageClick.bind(this)
    });
    
    // 挂载
    this.vueApp.mount(mountPoint);
  }

  /**
   * 应用主题
   */
  private async applyTheme(themeId: string): Promise<void> {
    if (!themeId || !this.container) return;
    
    // 通过Core获取主题
    const response = await this.core?.request({
      service: 'theme.get',
      payload: { themeId }
    });
    
    if (!response?.success) return;
    
    const theme = response.data;
    
    // 应用CSS变量
    const vars = theme.variables;
    if (vars?.colors) {
      this.container.style.setProperty(
        '--richtext-text-color', 
        vars.colors.text?.primary || '#333'
      );
      this.container.style.setProperty(
        '--richtext-link-color', 
        vars.colors.primary || '#0066cc'
      );
    }
    
    this.setState({ currentTheme: themeId });
  }

  /**
   * 绑定事件
   */
  private bindEvents(): void {
    if (!this.container) return;
    
    // 监听容器大小变化
    const resizeObserver = new ResizeObserver((entries) => {
      for (const entry of entries) {
        this.setState({ containerWidth: entry.contentRect.width });
      }
    });
    resizeObserver.observe(this.container);
  }

  /**
   * 解绑事件
   */
  private unbindEvents(): void {
    // ResizeObserver会在元素移除时自动断开
  }

  /**
   * 处理链接点击
   */
  private handleLinkClick(url: string): void {
    if (!this.options?.interactive) return;
    
    // 通过Core打开链接
    this.core?.request({
      service: 'system.openUrl',
      payload: { url, newWindow: true }
    });
  }

  /**
   * 处理图片点击
   */
  private handleImageClick(src: string): void {
    if (!this.options?.interactive) return;
    
    // 通过Core打开图片查看器
    this.core?.request({
      service: 'viewer.openImage',
      payload: { src }
    });
  }
}
```

```vue
<template>
  <div 
    class="chips-richtext-content"

import { computed, onMounted, onUnmounted } from 'vue';
import type { RenderOptions } from '../types';

// Props
const props = defineProps<{

  options?: RenderOptions;
}>();

// Emits

  (e: 'imageClick', src: string): void;

 * 处理后的内容

const processedContent = computed(() => {
  let html = props.content;
  
  // 添加样式类名
  html = html.replace(/<p>/g, '<p class="chips-richtext-paragraph">');
  html = html.replace(/<h1>/g, '<h1 class="chips-richtext-h1">');
  html = html.replace(/<h2>/g, '<h2 class="chips-richtext-h2">');
  html = html.replace(/<h3>/g, '<h3 class="chips-richtext-h3">');
  html = html.replace(/<h4>/g, '<h4 class="chips-richtext-h4">');
  html = html.replace(/<h5>/g, '<h5 class="chips-richtext-h5">');
  html = html.replace(/<h6>/g, '<h6 class="chips-richtext-h6">');
  html = html.replace(/<ul>/g, '<ul class="chips-richtext-ul">');
  html = html.replace(/<ol>/g, '<ol class="chips-richtext-ol">');
  html = html.replace(/<li>/g, '<li class="chips-richtext-li">');
  html = html.replace(/<blockquote>/g, '<blockquote class="chips-richtext-blockquote">');
  html = html.replace(/<a /g, '<a class="chips-richtext-link" ');
  html = html.replace(/<img /g, '<img class="chips-richtext-image" ');
  html = html.replace(/<code>/g, '<code class="chips-richtext-code">');
  
  return html;
});

/**
 * 处理点击事件
 */
function handleClick(event: MouseEvent): void {
  const target = event.target as HTMLElement;
  
  // 处理链接点击
  if (target.tagName === 'A') {
    event.preventDefault();
    const href = target.getAttribute('href');
    if (href) {
      emit('linkClick', href);
    }
    return;
  }
  
  // 处理图片点击
  if (target.tagName === 'IMG') {
    const src = target.getAttribute('src');
    if (src) {
      emit('imageClick', src);
    }
    return;
  }
}
</script>
```

## 5. HTML解析器

### 5.1 parser.ts

```typescript
/**
 * HTML解析器
 * 用于解析和处理富文本HTML内容
 */

/**
 * 解析HTML为DOM
 */
export function parseHtml(html: string): DocumentFragment {
  const template = document.createElement('template');
  template.innerHTML = html.trim();
  return template.content;
}

/**
 * 遍历DOM节点
 */
export function walkNodes(
  node: Node,
  callback: (node: Node) => void
): void {
  callback(node);
  
  let child = node.firstChild;
  while (child) {
    walkNodes(child, callback);
    child = child.nextSibling;
  }
}

/**
 * 提取纯文本
 */
export function extractText(html: string): string {
  const doc = parseHtml(html);
  return doc.textContent || '';
}

/**
 * 统计字数
 */
export function countWords(html: string): number {
  const text = extractText(html);
  // 移除空白字符后计数
  return text.replace(/\s/g, '').length;
}

/**
 * 检查是否为空内容
 */
export function isEmpty(html: string): boolean {
  const text = extractText(html);
  return text.trim().length === 0;
}

/**
 * 获取第一张图片
 */
export function getFirstImage(html: string): string | null {
  const match = html.match(/<img[^>]+src=["']([^"']+)["']/i);
  return match ? match[1] : null;
}

/**
 * 获取所有链接
 */
export function getAllLinks(html: string): string[] {
  const links: string[] = [];
  const regex = /<a[^>]+href=["']([^"']+)["']/gi;
  let match;
  while ((match = regex.exec(html)) !== null) {
    links.push(match[1]);
  }
  return links;
}
```

## 8. 响应式设计

### 8.1 容器宽度适配

```typescript
// 监听容器大小变化
private observeResize(): void {
  const resizeObserver = new ResizeObserver((entries) => {
    for (const entry of entries) {
      const width = entry.contentRect.width;
      this.setState({ containerWidth: width });
      this.applyResponsiveStyles(width);
    }
  });
  
  resizeObserver.observe(this.container);
}

// 应用响应式样式
private applyResponsiveStyles(width: number): void {
  const container = this.container;
  if (!container) return;
  
  // 移除所有断点类
  container.classList.remove(
    'chips-richtext--mobile',
    'chips-richtext--tablet',
    'chips-richtext--desktop'
  );
  
  // 添加对应断点类
  if (width < 480) {
    container.classList.add('chips-richtext--mobile');
  } else if (width < 768) {
    container.classList.add('chips-richtext--tablet');
  } else {
    container.classList.add('chips-richtext--desktop');
  }
}
```

## 9. 性能优化

### 9.1 图片懒加载

```typescript
private setupLazyLoading(): void {
  const images = this.container?.querySelectorAll('img');
  if (!images?.length) return;
  
  const observer = new IntersectionObserver((entries) => {
    entries.forEach(entry => {
      if (entry.isIntersecting) {
        const img = entry.target as HTMLImageElement;
        const src = img.dataset.src;
        if (src) {
          img.src = src;
          img.removeAttribute('data-src');
        }
        observer.unobserve(img);
      }
    });
  });
  
  images.forEach(img => {
    // 将src移到data-src
    const src = img.src;
    img.dataset.src = src;
    img.src = 'placeholder.png';
    observer.observe(img);
  });
}
```

### 10.1 单元测试

```typescript
import { describe, it, expect, beforeEach } from 'vitest';
import { RichTextRenderer } from '../renderer';

describe('RichTextRenderer', () => {
  let renderer: RichTextRenderer;
  let container: HTMLElement;
  
  beforeEach(() => {
    renderer = new RichTextRenderer();
    container = document.createElement('div');
    document.body.appendChild(container);
  });
  
  it('should render inline content', async () => {
    await renderer.render(
      {
        card_type: 'RichTextCard',
        content_source: 'inline',

      container,
      { mode: 'view' }

    expect(container.textContent).toContain('Hello World');

  it('should sanitize dangerous content', async () => {

      {
        card_type: 'RichTextCard',

      container,
      { mode: 'view' }

    expect(container.querySelector('script')).toBeNull();

      {
        card_type: 'RichTextCard',
        content_source: 'inline',

      container,
      { mode: 'view' }

    await renderer.update({

    });
    
    expect(container.textContent).toContain('After');
  });
  
  it('should destroy properly', async () => {
    await renderer.render(
      {
        card_type: 'RichTextCard',
        content_source: 'inline',
        content_text: '<p>Test</p>'
      },
      container,
      { mode: 'view' }
    );
    
    await renderer.destroy();
    
    expect(container.innerHTML).toBe('');
  });
});
```
