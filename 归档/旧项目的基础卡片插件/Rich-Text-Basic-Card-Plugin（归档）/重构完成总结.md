### 示例1：插件初始化

```typescript
// ❌ 改进前
async initialize(core: ChipsCore): Promise<void> {
  if (this.initialized) {
    console.warn('[RichTextCardPlugin] Already initialized');
    return;
  }
  this.core = core;
  // 注册插件服务
  // 实际会通过Core注册服务
  this.initialized = true;
  console.log('[RichTextCardPlugin] Initialized');
}

// ✅ 改进后
async initialize(core: ChipsCore): Promise<void> {
  if (this.initialized) {
    await this.logInfo('plugin.already_initialized');
    return;
  }
  this.core = core;
  await this.registerServices();  // 实际实现服务注册
  this.initialized = true;
  await this.logInfo('plugin.initialized');
}
```
