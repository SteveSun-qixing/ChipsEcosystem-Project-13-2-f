### 示例5：资源访问

```typescript
// ❌ 改进前
service: 'resource.read',
payload: {
  uri: this.config.content_file,
  encoding: 'utf-8',
}

// ✅ 改进后
service: 'resource.fetch',
payload: {
  uri: this.buildResourceUri(this.config.content_file),
  options: {
    as: 'text',
    encoding: 'utf-8',
    cache: true,  // 启用缓存
  },
}

private buildResourceUri(path: string): string {
  if (path.startsWith('http')) return `chips://network/${path}`;
  if (path.startsWith('/')) return `chips://local${path}`;
  return `chips://card/${cardId}/${path}`;
}
```
