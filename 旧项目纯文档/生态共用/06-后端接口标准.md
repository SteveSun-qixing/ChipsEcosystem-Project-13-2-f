## 1. 概述

### 1.1 后端模块定位

后端模块专注于**业务逻辑和数据处理**,不涉及界面展示:

- 处理业务规则
- 管理数据持久化
- 执行计算和转换
- 调用外部服务
- 不包含任何UI代码

### 1.2 后端与前端的关系

```
前端组件 → Core → 后端模块
         ↑        ↓
         └────────┘
         
前端: UI + 用户交互
后端: 业务逻辑 + 数据处理
Core: 路由 + 协调
```

### 1.3 技术无关原则

后端模块可以使用任何技术栈:
- JavaScript/TypeScript（推荐，与内核统一）
- Python
- Go
- Java
- C++

**唯一要求**: 实现标准接口，能通过内核路由器通信。薯片内核使用 TypeScript 实现，运行在 Electron 主进程中。插件通过 Bridge API（`window.chips.*`）调用后端服务。

```typescript
interface ModuleMetadata {

    };
```

### 2.2 服务注册

模块初始化时必须向Core注册提供的服务:

```typescript
interface ServiceRegistration {

interface ServiceDefinition {
  // 服务标识
  name: string;            // 服务名称,格式: "namespace.action"
  description: string;     // 服务描述
  
  // 接口定义
  input_schema: JSONSchema;   // 输入参数Schema
  output_schema: JSONSchema;  // 输出数据Schema

    enabled: boolean;
    ttl: number;           // 缓存有效期(秒)
  };
  rate_limit?: {
    requests: number;      // 请求数
    window: number;        // 时间窗口(秒)
  };
}
```

## 3. 服务处理器接口

### 3.1 处理器函数签名

```typescript
type ServiceHandler = (
  payload: any,           // 请求参数
  context: RequestContext // 请求上下文
) => Promise<ServiceResponse>;

interface RequestContext {
  request_id: string;     // 请求ID
  sender: string;         // 发送方模块ID
  timestamp: string;      // 请求时间
  permissions: string[];  // 已授予的权限
  user?: UserInfo;        // 用户信息(如有)
}

interface ServiceResponse {
  status: 'success' | 'error' | 'partial';
  data?: any;
  error?: {
    code: string;
    message: string;
    details?: any;
  };
  metadata?: {
    execution_time?: number;
    cache_hit?: boolean;
    [key: string]: any;
  };
}
```

```typescript
class CardManagerModule {
  metadata: ModuleMetadata = {
    id: "core:card-manager",
    name: "Card Manager",
    version: "1.0.0",
    description: "卡片管理模块",
    author: "Chips Core Team",
    license: "MIT",
    dependencies: {
      services: ["file-system", "resource-manager"]
    },
    protocol_version: {
      min: "1.0.0",
      max: "1.5.0"
    },
    required_permissions: [
      "card.read",
      "card.write",
      "resource.read",
      "resource.write"
    ]
  };
  
  // 注册服务
  getServiceDefinitions(): ServiceDefinition[] {
    return [
      {
        name: "card.read",
        description: "读取卡片内容",
        input_schema: {
          type: "object",
          required: ["card_id"],
          properties: {
            card_id: { type: "string", pattern: "^[0-9a-zA-Z]{10}$" },
            include_content: { type: "boolean", default: true },
            include_resources: { type: "boolean", default: false }
          }
        },
        output_schema: {
          type: "object",
          properties: {
            card: { $ref: "#/definitions/Card" }
          }
        },
        required_permissions: ["card.read"],
        handler: "CardManagerModule.readCard",
        timeout: 5000,
        cache: {
          enabled: true,
          ttl: 300
        }
      },
      {
        name: "card.write",
        description: "写入卡片内容",
        input_schema: {
          type: "object",
          required: ["card_id", "updates"],
          properties: {
            card_id: { type: "string" },
            updates: { type: "object" }
          }
        },
        output_schema: {
          type: "object",
          properties: {
            success: { type: "boolean" },
            modified_at: { type: "string", format: "date-time" }
          }
        },
        required_permissions: ["card.write"],
        handler: "CardManagerModule.writeCard",
        timeout: 10000
      }
    ];
  }
  
  // 处理器实现
  async readCard(
    payload: { card_id: string; include_content?: boolean; include_resources?: boolean },
    context: RequestContext
  ): Promise<ServiceResponse> {
    try {
      // 1. 验证参数
      if (!payload.card_id || payload.card_id.length !== 10) {
        return {
          status: 'error',
          error: {
            code: 'INVALID_CARD_ID',
            message: '卡片ID格式错误'
          }
        };
      }
      
      // 2. 读取卡片文件
      const cardPath = this.resolveCardPath(payload.card_id);
      const cardData = await this.fileSystem.read(cardPath);
      
      // 3. 解析卡片结构
      const card = await this.parseCardFile(cardData);
      
      // 4. 可选:加载基础卡片配置
      if (payload.include_content !== false) {
        card.content = await this.loadCardContent(payload.card_id);
      }
      
      // 5. 可选:加载资源列表
      if (payload.include_resources === true) {
        card.resources = await this.loadCardResources(payload.card_id);
      }
      
      // 6. 返回结果
      return {
        status: 'success',
        data: { card },
        metadata: {
          execution_time: Date.now() - startTime,
          cache_hit: false
        }
      };
      
    } catch (error) {
      return {
        status: 'error',
        error: {
          code: 'CARD_READ_FAILED',
          message: error.message,
          details: { card_id: payload.card_id }
        }
      };
    }
  }
  
  async writeCard(
    payload: { card_id: string; updates: any },
    context: RequestContext
  ): Promise<ServiceResponse> {
    try {
      // 1. 验证权限
      if (!context.permissions.includes('card.write')) {
        return {
          status: 'error',
          error: {
            code: 'PERMISSION_DENIED',
            message: '没有写入权限'
          }
        };
      }
      
      // 2. 读取现有卡片
      const card = await this.loadCard(payload.card_id);
      
      // 3. 应用更新
      const updatedCard = this.applyUpdates(card, payload.updates);
      
      // 4. 验证更新后的数据
      const validation = this.validateCard(updatedCard);
      if (!validation.valid) {
        return {
          status: 'error',
          error: {
            code: 'VALIDATION_FAILED',
            message: '数据验证失败',
            details: validation.errors
          }
        };
      }
      
      // 5. 保存卡片
      await this.saveCard(payload.card_id, updatedCard);
      
      // 6. 发布更新事件
      await this.publishEvent({
        event_type: 'card.updated',
        data: {
          card_id: payload.card_id,
          changes: payload.updates
        }
      });
      
      return {
        status: 'success',
        data: {
          success: true,
          modified_at: new Date().toISOString()
        }
      };
      
    } catch (error) {
      return {
        status: 'error',
        error: {
          code: 'CARD_WRITE_FAILED',
          message: error.message,
          details: { card_id: payload.card_id }
        }
      };
    }
  }
  
  // === 生命周期方法 ===
  
  async initialize(config: ModuleConfig): Promise<void> {
    this.config = config;
    this.fileSystem = await this.getService('file-system');
    this.resourceManager = await this.getService('resource-manager');
    // 初始化数据库连接等
  }
  
  async start(): Promise<void> {

  }
  
  async stop(): Promise<void> {
    // 停止后台任务
  }
  
  async destroy(): Promise<void> {
    // 清理资源
  }
  
  async healthCheck(): Promise<HealthStatus> {
    return {
      status: 'healthy',
      checks: {
        file_system: await this.fileSystem.isHealthy(),
        database: await this.database.ping()
      }
    };
  }
  
  // === 私有辅助方法 ===
  
  private resolveCardPath(cardId: string): string {
    // 解析卡片文件路径
    return `${this.config.cards_directory}/${cardId}.card`;
  }
  
  private async parseCardFile(data: Buffer): Promise<Card> {

  }

  }
```

## 6. 异步任务处理

### 6.1 任务接口

```typescript
interface TaskManager {
  // 创建任务
  createTask(
    taskType: string,
    payload: any,
    options?: TaskOptions
  ): Promise<Task>;
  
  // 查询任务状态
  getTask(taskId: string): Promise<Task>;
  
  // 取消任务
  cancelTask(taskId: string): Promise<void>;
  
  // 监听任务完成
  onTaskComplete(taskId: string, callback: (result: any) => void): void;
}

interface Task {
  id: string;
  type: string;
  status: 'pending' | 'running' | 'completed' | 'failed' | 'cancelled';
  payload: any;
  result?: any;
  error?: any;
  created_at: string;
  updated_at: string;
  progress?: number;
}

interface TaskOptions {
  priority?: 'low' | 'normal' | 'high';
  timeout?: number;
  retry?: {
    max_attempts: number;
    backoff: 'fixed' | 'exponential';
    delay: number;
  };
}
```

### 6.2 任务处理示例

```typescript
class MyBackendModule {
  async processLargeFile(fileId: string, context: RequestContext): Promise<ServiceResponse> {
    // 创建异步任务
    const task = await this.taskManager.createTask(
      'process_large_file',
      { file_id: fileId },
      {
        priority: 'high',
        timeout: 300000,  // 5分钟
        retry: {
          max_attempts: 3,
          backoff: 'exponential',
          delay: 1000
        }
      }
    );
    
    // 立即返回任务ID
    return {
      status: 'success',
      data: {
        task_id: task.id,
        status: 'pending',
        estimated_time: 60000
      }
    };
  }
  
  // 任务处理器
  async handleProcessLargeFileTask(task: Task): Promise<void> {
    const { file_id } = task.payload;
    
    try {
      // 执行处理...
      for (let i = 0; i < 100; i++) {
        // 处理步骤
        await this.processChunk(file_id, i);
        
        // 更新进度
        await this.taskManager.updateProgress(task.id, i + 1);
      }
      
      // 标记完成
      await this.taskManager.completeTask(task.id, {
        processed: true,
        file_id
      });
      
      // 发布完成事件
      await this.core.request({
        service: 'event.publish',
        payload: {
          event_type: 'task.completed',
          data: { task_id: task.id }
        }
      });
      
    } catch (error) {
      // 标记失败
      await this.taskManager.failTask(task.id, error);
    }
  }
}
```

## 8. 日志记录接口

### 8.1 日志接口

```typescript
interface Logger {
  // 日志级别
  debug(message: string, context?: any): void;
  info(message: string, context?: any): void;
  warn(message: string, context?: any): void;
  error(message: string, error?: Error, context?: any): void;
  
  // 结构化日志
  log(level: LogLevel, message: string, data?: any): void;
}

type LogLevel = 'debug' | 'info' | 'warn' | 'error';
```

### 8.2 日志使用示例

```typescript
class MyBackendModule {
  private logger: Logger;
  
  async readCard(payload: any, context: RequestContext): Promise<ServiceResponse> {
    this.logger.info('读取卡片', {
      card_id: payload.card_id,
      requester: context.sender,
      request_id: context.request_id
    });
    
    try {
      const card = await this.loadCard(payload.card_id);
      
      this.logger.debug('卡片读取成功', {
        card_id: payload.card_id,
        size: JSON.stringify(card).length
      });
      
      return { status: 'success', data: card };
      
    } catch (error) {
      this.logger.error('卡片读取失败', error, {
        card_id: payload.card_id
      });
      
      return {
        status: 'error',
        error: {
          code: 'CARD_READ_FAILED',
          message: error.message
        }
      };
    }
  }
}
```

### 10.2 错误处理示例

```typescript
class MyBackendModule {
  async readCard(payload: any, context: RequestContext): Promise<ServiceResponse> {
    try {
      // 参数验证
      if (!payload.card_id) {
        return this.errorResponse(
          ErrorCodes.MISSING_REQUIRED_FIELD,
          'card_id是必需参数',
          { field: 'card_id' }
        );
      }
      
      // 权限检查
      if (!context.permissions.includes('card.read')) {
        return this.errorResponse(
          ErrorCodes.PERMISSION_DENIED,
          '没有读取卡片的权限'
        );
      }
      
      // 读取卡片
      const card = await this.loadCard(payload.card_id);
      if (!card) {
        return this.errorResponse(
          ErrorCodes.RESOURCE_NOT_FOUND,
          `卡片 ${payload.card_id} 不存在`,
          { card_id: payload.card_id }
        );
      }
      
      return { status: 'success', data: { card } };
      
    } catch (error) {
      this.logger.error('读取卡片时发生错误', error);
      
      return this.errorResponse(
        ErrorCodes.INTERNAL_ERROR,
        '读取卡片失败',
        { original_error: error.message }
      );
    }
  }
  
  private errorResponse(code: string, message: string, details?: any): ServiceResponse {
    return {
      status: 'error',
      error: { code, message, details }
    };
  }
}
```

```typescript
async function getCardsWithTags(cardIds: string[]) {

    'SELECT * FROM cards WHERE id IN (?)',
    [cardIds]
  );
  const tags = await db.query(
    'SELECT * FROM tags WHERE card_id IN (?)',
    [cardIds]

  const tagsByCardId = groupBy(tags, 'card_id');

  }));
```

### 12.3 缓存策略

```typescript
class OptimizedModule {
  private cache: Map<string, { data: any; expires: number }> = new Map();
  
  async getData(key: string): Promise<any> {
    // 检查缓存
    const cached = this.cache.get(key);
    if (cached && cached.expires > Date.now()) {
      return cached.data;
    }
    
    // 加载数据
    const data = await this.loadData(key);
    
    // 写入缓存
    this.cache.set(key, {
      data,
      expires: Date.now() + 300000  // 5分钟
    });
    
    return data;
  }
}
```
