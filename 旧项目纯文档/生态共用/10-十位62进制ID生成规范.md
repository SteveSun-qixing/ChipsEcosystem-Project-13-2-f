**版本**: 1.0.0  
**更新时间**: 2026-01-30  
**状态**: 正式版

## 1. 概述

### 1.1 ID系统定位

Chips生态使用**10位62进制编号**作为所有实体的唯一标识符,包括:
- 卡片ID (Card ID)
- 箱子ID (Box ID)
- 基础卡片ID (Base Card ID)
- 用户ID (User ID)
- 其他需要唯一标识的实体

### 1.2 设计目标

1. **全局唯一**: 跨设备、跨时间保证唯一性
2. **紧凑性**: 10位字符长度,易于存储和传输
3. **可读性**: 使用字母和数字,易于阅读和输入
4. **无序性**: 不可预测,防止枚举攻击
5. **高效性**: 快速生成,无需中心化协调

## 2. 62进制编码

### 2.1 字符集

62进制使用以下62个字符:

```
0-9  (数字,10个字符)
a-z  (小写字母,26个字符)
A-Z  (大写字母,26个字符)

总计: 62个字符
```

**字符集定义**:
```
const BASE62_CHARS = '0123456789abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ';
```

### 2.3 容量计算

10位62进制可以表示的数量:

```
62^10 = 839,299,365,868,340,224
约等于 839 千万亿
```

**对比**:
- UUID (128位): 3.4 × 10^38
- 10位62进制: 8.4 × 10^17
- Snowflake (64位): 1.8 × 10^19

### 2.4 碰撞概率

假设已生成N个ID,生成第N+1个ID时的碰撞概率:

```
P(collision) ≈ N / 62^10

当N = 1,000,000 (100万):
P ≈ 0.000000119% (极低)

当N = 1,000,000,000 (10亿):
P ≈ 0.000119% (仍然很低)
```

## 3. ID生成算法

### 3.1 算法选择

推荐使用**加密安全的随机数生成器**:

```typescript
function generateId(): string {
  // 使用加密安全的随机数
  const bytes = crypto.getRandomValues(new Uint8Array(8));
  
  // 转换为62进制
  let value = 0n;
  for (let i = 0; i < bytes.length; i++) {
    value = (value << 8n) | BigInt(bytes[i]);
  }
  
  // 编码为10位62进制字符串
  return encodeBase62(value, 10);
}
```

### 3.2 编码函数

```typescript
const BASE62_CHARS = '0123456789abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ';

function encodeBase62(value: bigint, length: number): string {
  let result = '';
  
  while (value > 0n) {
    const remainder = Number(value % 62n);
    result = BASE62_CHARS[remainder] + result;
    value = value / 62n;
  }
  
  // 左填充到指定长度
  return result.padStart(length, '0');
}
```

### 3.3 解码函数

```typescript
function decodeBase62(str: string): bigint {
  let value = 0n;
  
  for (let i = 0; i < str.length; i++) {
    const char = str[i];
    const index = BASE62_CHARS.indexOf(char);
    
    if (index === -1) {
      throw new Error(`Invalid character: ${char}`);
    }
    
    value = value * 62n + BigInt(index);
  }
  
  return value;
}
```

## 4. ID验证

### 4.1 验证规则

有效的ID必须满足:
1. **长度**: 恰好10个字符
2. **字符集**: 只包含`0-9a-zA-Z`
3. **非空**: 不能为全0

### 4.2 验证函数

```typescript
function validateId(id: string): boolean {
  // 长度检查
  if (id.length !== 10) {
    return false;
  }
  
  // 字符集检查
  const pattern = /^[0-9a-zA-Z]{10}$/;
  if (!pattern.test(id)) {
    return false;
  }
  
  // 非全0检查
  if (id === '0000000000') {
    return false;
  }
  
  return true;
}
```

### 4.3 TypeScript类型定义

```typescript
// 品牌类型,确保类型安全
type CardId = string & { readonly __brand: 'CardId' };
type BoxId = string & { readonly __brand: 'BoxId' };
type BaseCardId = string & { readonly __brand: 'BaseCardId' };

function createCardId(id: string): CardId {
  if (!validateId(id)) {
    throw new Error('Invalid card ID');
  }
  return id as CardId;
}

// 使用
const cardId: CardId = createCardId('a1B2c3D4e5');
```

## 5. ID类型前缀(可选)

### 5.1 前缀设计

虽然ID本身不包含类型信息,但在某些场景可以使用前缀区分类型:

```typescript
type IdPrefix = 'card' | 'box' | 'base' | 'user';

function formatIdWithPrefix(type: IdPrefix, id: string): string {
  return `${type}_${id}`;
}

// 示例
const cardId = formatIdWithPrefix('card', 'a1B2c3D4e5');
// 结果: "card_a1B2c3D4e5"
```

### 5.2 解析带前缀ID

```typescript
function parseIdWithPrefix(prefixedId: string): {
  type: IdPrefix;
  id: string;
} {
  const parts = prefixedId.split('_');
  
  if (parts.length !== 2) {
    throw new Error('Invalid prefixed ID format');
  }
  
  const [type, id] = parts;
  
  if (!validateId(id)) {
    throw new Error('Invalid ID');
  }
  
  return { type: type as IdPrefix, id };
}
```

## 6. 批量生成

### 6.1 批量生成函数

```typescript
function generateIds(count: number): string[] {
  const ids = new Set<string>();
  
  while (ids.size < count) {
    const id = generateId();
    ids.add(id);
  }
  
  return Array.from(ids);
}

// 使用
const cardIds = generateIds(100);  // 生成100个唯一ID
```

### 6.2 碰撞检测

```typescript
function generateUniqueId(existingIds: Set<string>): string {
  let attempts = 0;
  const maxAttempts = 10;
  
  while (attempts < maxAttempts) {
    const id = generateId();
    
    if (!existingIds.has(id)) {
      return id;
    }
    
    attempts++;
  }
  
  // 极端情况下的后备方案
  throw new Error('Failed to generate unique ID after multiple attempts');
}
```

## 7. 性能优化

### 7.1 预生成池

```typescript
class IdPool {
  private pool: string[] = [];
  private readonly minSize = 100;
  private readonly maxSize = 1000;
  
  constructor() {
    this.refill();
  }
  
  async getId(): Promise<string> {
    if (this.pool.length < this.minSize) {
      this.refill();
    }
    
    return this.pool.pop()!;
  }
  
  private refill(): void {
    const count = this.maxSize - this.pool.length;
    const newIds = generateIds(count);
    this.pool.push(...newIds);
  }
}

// 使用
const idPool = new IdPool();
const cardId = await idPool.getId();
```

### 7.2 性能基准

```typescript
// 单次生成性能测试
function benchmarkGeneration(iterations: number): number {
  const start = performance.now();
  
  for (let i = 0; i < iterations; i++) {

  console.log(`Generated ${iterations} IDs in ${duration}ms`);
  console.log(`Rate: ${rate.toFixed(0)} IDs/second`);
  
  return rate;
}
```

## 8. 安全性考虑

### 8.1 使用加密安全随机数

**✅ 正确**: 使用加密安全的随机数生成器

```typescript
// Node.js
import { randomBytes } from 'crypto';

function generateId(): string {
  const bytes = randomBytes(8);
  // 转换逻辑...
}

// 浏览器
function generateId(): string {
  const bytes = crypto.getRandomValues(new Uint8Array(8));
  // 转换逻辑...
}
```

**❌ 错误**: 使用`Math.random()`

```typescript
// 不要这样做! Math.random()不够安全
function generateId(): string {
  const value = Math.random() * (62 ** 10);
  return encodeBase62(BigInt(Math.floor(value)), 10);
}
```

### 8.2 防止时序攻击

ID生成应该是恒定时间操作,不泄露任何信息:

```typescript
// 好的实践: 使用固定大小的随机数
function generateId(): string {
  const bytes = crypto.getRandomValues(new Uint8Array(8));
  // ...
}

// 避免: 使用时间戳等可预测信息
function generateId(): string {
  const timestamp = Date.now();  // 可预测
  // ...
}
```

### 9.2 Python

```python
import secrets
import string

BASE62_CHARS = string.digits + string.ascii_lowercase + string.ascii_uppercase

def generate_id() -> str:
    # 生成8字节随机数
    value = int.from_bytes(secrets.token_bytes(8), 'big')
    
    # 转换为62进制
    return encode_base62(value, 10)

def encode_base62(value: int, length: int) -> str:
    if value == 0:
        return '0' * length
    
    result = []
    while value > 0:
        result.append(BASE62_CHARS[value % 62])
        value //= 62
    
    result.reverse()
    return ''.join(result).rjust(length, '0')
```

### 9.3 Rust

```rust
use rand::Rng;

const BASE62_CHARS: &[u8] = b"0123456789abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ";

pub fn generate_id() -> String {
    let mut rng = rand::thread_rng();
    let value: u64 = rng.gen();
    
    encode_base62(value, 10)
}

fn encode_base62(mut value: u64, length: usize) -> String {
    let mut result = Vec::new();
    
    while value > 0 {
        result.push(BASE62_CHARS[(value % 62) as usize]);
        value /= 62;
    }
    
    result.reverse();
    
    let mut id = String::from_utf8(result).unwrap();
    while id.len() < length {
        id.insert(0, '0');
    }
    
    id
}
```

### 9.4 Go

```go
package chips

import (
    "crypto/rand"
    "math/big"
)

const base62Chars = "0123456789abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ"

func GenerateID() (string, error) {
    // 生成8字节随机数
    bytes := make([]byte, 8)
    _, err := rand.Read(bytes)
    if err != nil {
        return "", err
    }
    
    value := new(big.Int).SetBytes(bytes)
    return encodeBase62(value, 10), nil
}

func encodeBase62(value *big.Int, length int) string {
    if value.Sign() == 0 {
        return "0000000000"
    }
    
    result := ""
    base := big.NewInt(62)
    zero := big.NewInt(0)
    
    for value.Cmp(zero) > 0 {
        mod := new(big.Int)
        value.DivMod(value, base, mod)
        result = string(base62Chars[mod.Int64()]) + result
    }
    
    for len(result) < length {
        result = "0" + result
    }
    
    return result
}
```

**✅ 推荐**: 在创建实体时生成

```typescript
class CardManager {
```

## 12. 常见问题

### Q1: 为什么不使用UUID?

**回答**: 
- UUID (128位) 过长,不适合显示和手动输入
- 10位62进制已足够保证唯一性
- 更紧凑,节省存储空间

### Q2: 如何确保分布式系统中的唯一性?

**回答**:
- 使用加密安全的随机数生成器
- 碰撞概率极低(见2.4节)
- 可以在保存前检查是否已存在

### Q3: ID可以重用吗?

**回答**:
- 不建议重用ID
- 删除的实体ID应该永久废弃
- 避免引用混乱和数据不一致

### Q4: 如何迁移现有的ID系统?

**回答**:
```typescript
// 迁移策略
function migrateId(oldId: string): string {
  // 方案1: 创建映射表
  if (idMapping.has(oldId)) {
    return idMapping.get(oldId)!;
  }
  
  // 方案2: 生成新ID并记录映射
  const newId = generateId();
  idMapping.set(oldId, newId);
  return newId;
}
```

## 附录 A: 完整实现示例

```typescript
/**
 * 10位62进制ID生成器
 * 完整实现,包含生成、验证、编解码
 */

const BASE62_CHARS = '0123456789abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ';

export class IdGenerator {
  /**
   * 生成10位62进制ID
   */
  static generate(): string {
    if (typeof crypto !== 'undefined' && crypto.getRandomValues) {
      // 浏览器环境
      const bytes = crypto.getRandomValues(new Uint8Array(8));
      let value = 0n;
      for (const byte of bytes) {
        value = (value << 8n) | BigInt(byte);
      }
      return this.encodeBase62(value, 10);
    } else {
      // Node.js环境
      const crypto = require('crypto');
      const bytes = crypto.randomBytes(8);
      let value = 0n;
      for (const byte of bytes) {
        value = (value << 8n) | BigInt(byte);
      }
      return this.encodeBase62(value, 10);
    }
  }
  
  /**
   * 验证ID格式
   */
  static validate(id: string): boolean {
    if (id.length !== 10) return false;
    if (!/^[0-9a-zA-Z]{10}$/.test(id)) return false;
    if (id === '0000000000') return false;
    return true;
  }
  
  /**
   * 编码为62进制
   */
  static encodeBase62(value: bigint, length: number): string {
    if (value === 0n) {
      return '0'.repeat(length);
    }
    
    let result = '';
    while (value > 0n) {
      result = BASE62_CHARS[Number(value % 62n)] + result;
      value = value / 62n;
    }
    
    return result.padStart(length, '0');
  }
  
  /**
   * 解码62进制
   */
  static decodeBase62(str: string): bigint {
    let value = 0n;
    
    for (const char of str) {
      const index = BASE62_CHARS.indexOf(char);
      if (index === -1) {
        throw new Error(`Invalid character: ${char}`);
      }
      value = value * 62n + BigInt(index);
    }
    
    return value;
  }
  
  /**
   * 批量生成
   */
  static generateBatch(count: number): string[] {
    const ids = new Set<string>();
    while (ids.size < count) {
      ids.add(this.generate());
    }
    return Array.from(ids);
  }
}

// 使用示例
const cardId = IdGenerator.generate();
console.log(cardId);  // 输出: a1B2c3D4e5

const isValid = IdGenerator.validate(cardId);
console.log(isValid);  // 输出: true

const batchIds = IdGenerator.generateBatch(100);
console.log(batchIds.length);  // 输出: 100
```
